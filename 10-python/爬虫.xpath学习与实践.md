selects = tree.xpath('//div[@class="answer"]')
options = []
for key, values in enumerate(selects):
    item = values.xpath('.//span[@class="el-radio__label"]/span/text() | .//span[@class="el-checkbox__label"]/span/text()')
    itemStr = ';\n'.join(item)
    options.append(itemStr)
print(options)
